    通过上图，我们可以看到一个简单的服务发现并调用的机制。

    配置member服务

    在pom.xml中引入Nacos-Discovery

            <!-- https://mvnrepository.com/artifact/com.alibaba.cloud/spring-cloud-starter-alibaba-nacos-discovery -->
        <dependency>
            <groupId>com.alibaba.cloud</groupId>
            <artifactId>spring-cloud-starter-alibaba-nacos-discovery</artifactId>
            <version>2.1.0.RELEASE</version>
        </dependency>

    在bootstrap.yml中配置Nacos地址

    //注意这里只使用的NacosDiscovery没有使用NacosConfig。
server:
  port: 9991
spring:
  application:
    name: user
  cloud:
    nacos:
      discovery:
        server-addr: 127.0.0.1:8848

    通过Cloud原生注解@EnableDiscoveryClient开启服务发现

    @SpringBootApplication
@EnableDiscoveryClient
public class WhyApplication {

    public static void main(String[] args) {
        SpringApplication.run(WhyApplication.class, args);
    }

}

    在mamber服务中调用order服务

    配置RestTemplate

    RestTemplate是Spring提供的全家桶的一员，它的底层是对HttpClient进行了封装。

    @Component
public class RestTemplateConfig {

    @Bean
    @LoadBalanced
    public RestTemplate restTemplate(){
        return new RestTemplate();
    }

}

    调用order服务

    @RestController
public class UserController {

    @Autowired
    private RestTemplate restTemplate;

    @GetMapping("/get/order")
    public String getOrder(){
        return restTemplate.getForObject("http://order/list/",String.class);
    }

}
