    RestTemplateжҳҜSpringжҸҗдҫӣзҡ„е…Ёе®¶жЎ¶зҡ„дёҖе‘ҳпјҢе®ғзҡ„еә•еұӮжҳҜеҜ№HttpClientиҝӣиЎҢдәҶе°ҒиЈ…гҖӮ

public class RestTemplateConfig {

    @Bean
    @LoadBalanced
    public RestTemplate restTemplate(){
        return new RestTemplate();
    }

}
