//еҶҷдёҖдёӘз®ҖеҚ•зҡ„еҚ•дҫӢ
class Test{

    private static Test t = new Test();
    public static int counter1;
    public static int counter2 = 0;

    private Test() {
        counter1 ++;
        counter2 ++;
    }

    public static Test getInstance(){
        return t;
    }
}

    д»ҺзЎ¬зӣҳеҲ°еҶ…еӯҳ

      зұ»зҡ„еҠ иҪҪйЎәеәҸдё»иҰҒеҲҶдёәпјҡ

    • еҠ иҪҪпјҡд»ҺзЎ¬зӣҳе°ҶдәҢиҝӣеҲ¶ж–Үд»¶еҠ иҪҪеҲ°еҶ…еӯҳпјҢе°ҶиҜ»еҲ°еҲ°ж•°жҚ®ж”ҫеҲ°JVMж–№жі•еҢәдёӯпјҢз„¶еҗҺеңЁе ҶдёҠеҲӣе»әдёҖдёӘClassеҜ№иұЎз”ЁжқҘе°ҒиЈ…иҝҷдёӘзұ»еңЁж–№жі•еҢәеҲ°ж•°жҚ®з»“жһ„гҖӮд№ҹе°ұжҳҜдёҖдёӘй•ңеғҸеҜ№иұЎпјҢжҜҸдёҖдёӘзұ»йғҪжңүдёҖдёӘиҮӘе·ұеҲ°дё“еұһClassеҜ№иұЎпјҢжүҖд»ҘеңЁеҸҚе°„дёӯжҲ‘д»¬еҸҜд»ҘдҪҝз”ЁClassеҜ№иұЎжқҘиҺ·еҸ–зұ»еҲ°е…ЁйғЁж•°жҚ®гҖӮ
    • иҝһжҺҘпјҡ
      • йӘҢиҜҒпјҡдҝқиҜҒиў«еҠ иҪҪзұ»еҲ°жӯЈзЎ®жҖ§гҖӮйӘҢиҜҒз»“жһ„гҖҒиҜӯд№үгҖҒеӯ—иҠӮз ҒгҖҒе…је®№жҖ§зӯүгҖӮ
      • еҮҶеӨҮпјҡдёәзұ»еҲ°йқҷжҖҒеҸҳйҮҸеҲҶй…ҚеҶ…еӯҳпјҢе№¶и®ҫзҪ®й»ҳи®ӨеҖјгҖӮ
      • и§ЈжһҗпјҡжҠҠзұ»дёӯзҡ„з¬ҰеҸ·еј•з”ЁжҚўжҲҗжҲҗзӣҙжҺҘеј•з”ЁгҖӮ
    • еҲқе§ӢеҢ–пјҡдёәзұ»зҡ„йқҷжҖҒеҸҳйҮҸиөӢжӯЈзЎ®зҡ„еҖјгҖӮ

    еҲқе§ӢеҢ–

      жүҖжңүзҡ„JVMзұ»еҝ…йЎ»еңЁиў«зЁӢеәҸйҰ–ж¬Ўдё»еҠЁдҪҝз”Ёж—¶жүҚеҲқе§ӢеҢ–пјҢдҪҶжҳҜеҠ иҪҪе’ҢиҝһжҺҘйҳ¶ж®өе·Із»ҸеҸ‘з”ҹгҖӮ
      иҝҷйҮҢжҲ‘д»¬иҰҒиҜҙжҳҺдёҖдёӢдё»еҠЁдҪҝз”ЁеңәжҷҜпјҡ

    • newеҜ№иұЎпјҢеҪ“жҲ‘д»¬жүӢеҠЁеҲӣе»әдёҖдёӘеҜ№иұЎж—¶гҖӮ
    • и®ҝй—®жҹҗдёӘзұ»еҲ°йқҷжҖҒеҸҳйҮҸгҖӮ
    • и°ғз”ЁжҹҗдёӘзұ»еҲ°йқҷжҖҒж–№жі•гҖӮ
    • йҖҡиҝҮClassзұ»иҝӣиЎҢеҸҚе°„ж“ҚдҪңгҖӮ
    • еҲқе§ӢеҢ–дёҖдёӘзұ»еҲ°еӯҗзұ»гҖӮ
    • JVMеҗҜеҠЁж—¶йҖҡиҝҮеҸӮж•°ж ҮжҳҺдёәй»ҳи®ӨеҗҜеҠЁеҲ°зұ»гҖӮ

      жңүдё»еҠЁе°ұжңүиў«еҠЁпјҢиў«еҠЁдҪҝз”Ёе№¶дёҚдјҡи§ҰеҸ‘еҲқе§ӢеҢ–пјҡ

    • еҪ“жҲ‘д»¬йҖҡиҝҮеӯҗзұ»еҲ°йқҷжҖҒж–№жі•иҺ·еҸ–зҲ¶зұ»еҲ°йқҷжҖҒеҸҳйҮҸгҖӮ
    • еҲӣе»әж•°з»„еҜ№иұЎгҖӮE.g.  Student[] s = new Student[1]гҖӮ

    зұ»еҠ иҪҪеҷЁеҜ№иұЎ

    JVMйҖҡиҝҮзұ»еҠ иҪҪеҜ№иұЎжқҘиҜ»еҸ–.classж–Үд»¶пјҢиҝҷйҮҢжңүдёҖдёӘйҮҚиҰҒеҲ°жҰӮеҝөпјҡеҸҢдәІе§”жҙҫжңәеҲ¶

    зұ»еҠ иҪҪеҷЁеҲ°зұ»еһӢ

    • Bootstrap ClassloaderпјҡJVMжңҖйЎ¶зә§еҠ иҪҪеҷЁпјҢеҸӘиҙҹиҙЈеҠ иҪҪжңҖж ёеҝғеҲ°зұ»пјҢдёҚжҸҗдҫӣз»ҷејҖеҸ‘иҖ…дҪҝз”ЁгҖӮ
    • Extension ClassloaderпјҡиҙҹиҙЈеҠ иҪҪжү©еұ•зұ»еә“пјҢеҠ иҪҪ/lib/extзӣ®еҪ•дёӢеҲ°jarгҖӮ
    • Application ClassloaderпјҡеӨҚжқӮеҠ иҪҪclasspathдёӢйқўеҲ°жүҖжңүjarе’Ңclassж–Үд»¶пјҢд№ҹе°ұжҳҜиҜҙжҲ‘д»¬еҲ°дёҡеҠЎд»Јз ҒйғҪжҳҜйҖҡиҝҮе®ғжқҘеҠ иҪҪеҲ°JVMдёӯеҲ°гҖӮ
